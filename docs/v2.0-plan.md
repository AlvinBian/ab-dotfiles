# ab-dotfiles v2.0.0 完整方案

> 目標：極簡流程 + 全程進度感知 + 零缺陷 + 完整 Claude 配置 + 專案級部署

## 一、流程重設計

### 現狀問題

首次安裝 14 次互動，大部分是「確認預選」— 用戶不知道選什麼，只能按確認。
本質上只有 2 個真正需要用戶決策的問題：
1. 分析哪些 repos？
2. 最終安裝這些東西 OK 嗎？

### 新流程：首次 3 步，重入 1 步

| 場景 | v1.1.0 | v2.0.0 |
|------|--------|--------|
| 首次安裝 | 14 次互動 | **3 次** |
| 有 session | 8 次 | **1 次（Enter 重裝）** |
| --quick | 0 次 | 0 次 |
| 調整細項 | 必經 | 只在選「調整」時才展開 |

## 二、全程進度感知設計

### 原則

- 每個操作都有進度條（確定性）或 spinner（不確定性）
- 進度條顯示：百分比 + [N/N] + 當前項 + ETA（>3s 時）
- 階段之間有標題分隔
- 背景任務有提示
- 零死區：用戶永遠知道在做什麼、做了多少、還要多久

### 新依賴

```bash
pnpm add listr2    # 任務列表 + 進度 + 子任務 + 並行 + 計時
pnpm remove ora    # listr2 內建 spinner，不再需要
```

import { Listr } from 'listr2'
import pc from 'picocolors'

/**
 * 階段標題
 */
export function phaseHeader(title, step, total) {
  const prefix = step ? `Step ${step}/${total} — ` : ''
  console.log(`\n  ${'═'.repeat(52)}`)
  console.log(`  📍 ${prefix}${title}`)
  console.log(`  ${'═'.repeat(52)}\n`)
}

/**
 * 建立任務列表（順序執行）
 *
 * @param {Array<{ title, task, skip? }>} tasks
 * @param {Object} opts - listr2 options
 */
export function createTaskList(tasks, opts = {}) {
  return new Listr(tasks, {
    concurrent: false,
    exitOnError: false,
    rendererOptions: {
      showTimer: true,
      collapseSubtasks: false,
      showSubtasks: true,
      suffixSkips: true,
      ...opts.rendererOptions,
    },
    ...opts,
  })
}

/**
 * 建立並行任務列表
 */
export function createConcurrentTasks(tasks, opts = {}) {
  return new Listr(tasks, {
    concurrent: true,
    exitOnError: false,
    rendererOptions: {
      showTimer: true,
      collapseSubtasks: false,
      ...opts.rendererOptions,
    },
    ...opts,
  })
}

/**
 * 安裝主流程（7 子任務順序執行，各自可含子任務）
 *
 * 效果：
 *   ✔ [1/7] 備份 [1.2s]
 *   ✔ [2/7] 全局配置 [0.8s]
 *     ✔ commands (7)
 *     ✔ agents (8)
 *     ✔ rules (2)
 *     ✔ settings.json
 *     ✔ keybindings.json

 *   ◌ [4/7] CLAUDE.md
 *   ◌ [5/7] ECC + Stacks
 *   ◌ [6/7] Plugin
 *   ◌ [7/7] zsh 模組
 */
export function createInstallRunner(plan) {
  return createTaskList([
    {
      title: '[1/7] 備份',
      task: async (ctx, task) => {
        // 備份邏輯
        task.output = `${ctx.backupCount} 項`
      },
    },
    {
      title: '[2/7] 全局配置',
      task: (ctx, task) => task.newListr([
        { title: 'commands', task: async () => { /* 安裝 7 commands */ } },
        { title: 'agents', task: async () => { /* 安裝 8 agents */ } },
        { title: 'rules', task: async () => { /* 安裝 2 rules */ } },
        { title: 'settings.json (permissions + hooks)', task: async () => { /* 合併 */ } },
        { title: 'keybindings.json', skip: () => ctx.keybindingsExists ? '已有自訂快捷鍵' : false, task: async () => {} },
      ]),
    },
    {
      title: '[3/7] 專案配置',
      task: (ctx, task) => task.newListr(
        plan.projects.filter(p => p.localPath).map(p => ({
          title: `${p.role === 'main' ? '⭐' : '🔄'} ${p.repo.split('/')[1]}`,
          task: async (_, subtask) => {
            subtask.output = p.role === 'main' ? '完整配置' : '精簡配置'
            // 注入 .claude/
          },
        }))
      ),
    },
    {
      title: '[4/7] CLAUDE.md',
      task: (ctx, task) => task.newListr(
        plan.projects.filter(p => p.localPath).map(p => ({
          title: p.repo.split('/')[1],
          task: async (_, subtask) => {
            subtask.output = p.role === 'main' ? 'AI 生成（完整版）' : '模板（精簡版）'
            // AI 生成 CLAUDE.md
          },
        })),
        { concurrent: true }  // AI 生成可並行
      ),
    },
    {
      title: '[5/7] ECC + Stacks',
      task: (ctx, task) => task.newListr([
        { title: `ECC (${plan.ecc.length})`, task: async () => {} },
        { title: `Stacks (${plan.techStacks.length})`, task: async () => {} },
      ], { concurrent: true }),
    },
    {
      title: '[6/7] Plugin',
      task: (ctx, task) => task.newListr([
        { title: 'ab-claude-dev.plugin', task: async () => {} },
        { title: 'ab-slack-message.plugin', task: async () => {} },

    },

/**
 * 分析階段（進度條替代）
 *
 * 效果：
 *   ✔ Repos + ECC [5.8s]
 *   ↓ AI 分類 [18.4s]
 *     ✔ kkday-b2c-web [8.2s]
 *     ✔ kkday-member-ci [12.1s]
 *     ◼ web-design-system [cache]
 *   ✔ ECC 規則匹配 — 41 個推薦 [<1s]
 *   ✔ 偵測本機路徑 — 3/4 找到 [0.8s]
 *   ✔ 生成計畫 [<1s]
 */
export function createAnalyzeRunner(repos, sources) {
  return createTaskList([
    {
      title: `Repos + ECC fetch`,
      task: async (ctx, task) => {
        // 並行 fetch
        task.output = `${repos.length} repos + ${sources.length} ECC`
      },
    },
    {
      title: 'AI 分類',
      task: (ctx, task) => task.newListr(
        repos.map(r => ({
          title: r.split('/')[1],
          task: async (_, subtask) => {
            // per-repo AI 分類
            subtask.output = ctx.fromCache ? 'cache' : `$${ctx.cost}`
          },
        })),
        { concurrent: true }  // AI 並行
      ),
    },
    {
      title: 'ECC 規則匹配',
      task: async (ctx, task) => {
        task.output = `${ctx.eccCount} 個推薦`
      },
    },
    {
      title: '偵測本機路徑',
      task: async (ctx, task) => {
        task.output = `${ctx.foundCount}/${repos.length} 找到`
      },
    },
    {
      title: '生成安裝計畫',
      task: async (ctx, task) => {
        task.output = '就緒'
      },
    },
  ])
}

### 8.1 配置衝突策略

| 配置檔 | 策略 | 邏輯 |
|--------|------|------|
| ~/.claude/settings.json | **merge** | 讀取已有 → 深度合併 permissions/hooks → 保留用戶自訂 + 追加新的 |
| ~/.claude/keybindings.json | **skip if exists** | 已存在則不覆蓋，只提示「已有自訂快捷鍵」 |
| repo/.claude/ | **backup + inject** | 已有 .claude/ 先備份到 dist/backup/，再注入新配置 |
| repo/CLAUDE.md | **skip if exists** | 已有 CLAUDE.md 不覆蓋（可能是用戶或團隊手寫的） |

```javascript
// lib/deploy/deploy-global.mjs — settings.json 合併邏輯
function mergeSettings(existing, template) {
  const merged = { ...existing }
  // permissions: 合併（去重）
  merged.permissions = {
    allow: [...new Set([...(existing.permissions?.allow || []), ...template.permissions.allow])],
    deny: [...new Set([...(existing.permissions?.deny || []), ...template.permissions.deny])],
  }
  // hooks: 按 event 合併（不覆蓋已有的同 matcher hook）
  merged.hooks = mergeHooks(existing.hooks || {}, template.hooks)
  // model/effort/env: 只在未設定時寫入
  if (!existing.model) merged.model = template.model
  if (!existing.effortLevel) merged.effortLevel = template.effortLevel
  if (!existing.autoMemoryEnabled) merged.autoMemoryEnabled = template.autoMemoryEnabled
  if (!existing.env) merged.env = template.env
  return merged
}
```

### 8.2 Spotlight fallback

```javascript
// lib/repo-detect.mjs
export async function detectLocalRepos(repos) {
  const results = {}
  for (const repo of repos) {
    const name = repo.fullName.split('/')[1]
    // macOS: Spotlight（<0.1s per repo）
    if (process.platform === 'darwin') {
      try {
        const dirs = execSync(
          `mdfind "kMDItemFSName == '${name}' && kMDItemContentType == public.folder" -onlyin ~`,
          { encoding: 'utf8', timeout: 5000 }
        ).trim().split('\n').filter(Boolean)
        for (const dir of dirs) {
          const remote = execSync(`git -C "${dir}" remote get-url origin 2>/dev/null`, { encoding: 'utf8' }).trim()
          if (remote.includes(name)) { results[repo.fullName] = dir; break }
        }
      } catch {}
    }
    // fallback: find（較慢，跨平台）
    if (!results[repo.fullName]) {
      try {
        const dirs = execSync(
          `find ~ -maxdepth 4 -name "${name}" -type d 2>/dev/null | head -5`,
          { encoding: 'utf8', timeout: 10000 }
        ).trim().split('\n').filter(Boolean)
        for (const dir of dirs) {
          try {
            const remote = execSync(`git -C "${dir}" remote get-url origin 2>/dev/null`, { encoding: 'utf8' }).trim()
            if (remote.includes(name)) { results[repo.fullName] = dir; break }
          } catch {}
        }
      } catch {}
    }
  }
  return results
}
```

### 8.3 錯誤恢復

| 失敗場景 | 恢復策略 |
|----------|---------|
| Spotlight 偵測失敗 | fallback find → 都失敗則生成到 dist/projects/ + 提示手動複製 |
| AI CLAUDE.md 生成失敗 | 用靜態模板 fallback（不含 AI 分析段落）|
| AI 分類超時 | 用快取 / 跳過該 repo / 用語言偵測替代 |
| repo 目錄唯讀 | 生成到 dist/projects/{org}/{repo}/ + 提示 |
| settings.json 合併失敗 | 備份原始 → 寫入模板 → 提示用戶手動合併 |
| 網路斷線 | 用快取完成能做的，跳過需網路的（AI/ECC/GitHub） |
| zsh 安裝中途失敗 | 記錄已完成的模組，下次 --quick 只裝剩餘的 |
